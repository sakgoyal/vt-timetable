A python module for scraping the Virginia Tech Timetable of Classes.

This module makes it easy to get data from the Virginia Tech Timetable of Classes using python. The most important components of the module are the search_timetable function, which directly searches the timetable with the provided parameters, and the Course class, which contains data about a course. Additionally, get_crn makes it easier to search for a specific course. The two other functions, get_semesters and get_subjects, make it possible to check which search parameters are valid.

132class Course:
133    """Represents a course at VT.
134
135    This class is used as a container for data about courses at Virginia Tech.
136    """
137
138    _section_type_dct = {'I': SectionType.INDEPENDENT_STUDY,
139                         'B': SectionType.LAB,
140                         'L': SectionType.LECTURE,
141                         'C': SectionType.RECITATION,
142                         'R': SectionType.RESEARCH,
143                         'O': SectionType.ONLINE}
144    _modality_dct = {'Face-to-Face Instruction': Modality.IN_PERSON,
145                     'Hybrid (F2F & Online Instruc.)': Modality.HYBRID,
146                     'Online with Synchronous Mtgs.': Modality.ONLINE_SYNC,
147                     'Online: Asynchronous': Modality.ONLINE_ASYNC}
148    _day_dct = {'M': Day.MONDAY, 'T': Day.TUESDAY, 'W': Day.WEDNESDAY,
149                'R': Day.THURSDAY, 'F': Day.FRIDAY, 'S': Day.SATURDAY,
150                'U': Day.SUNDAY, '(ARR)': Day.ARRANGED}
151
152    def __init__(self, year: str, semester: Semester,
153                 timetable_data: pandas.core.series.Series,
154                 extra_class_data: pandas.core.series.Series) -> None:
155        """ A constructor for `Course`.
156
157        Args:
158                year:
159                        string representing the year in which the course is taking
160                        place.
161                semester:
162                        `Semester` representing the semester in which the course is
163                        taking place.
164                timetable_data:
165                        `pandas.Series` representing the data scraped from the
166                        timetable.
167                extra_class_data:
168                        Optional `pandas.Series` representing the days and times of
169                        additional classes scraped from the timetable.
170        """
171
172        subject, code = re.match(r'(.+)-(.+)', timetable_data[1]).group(1, 2)
173
174        if semester == Semester.SUMMER:
175            name = re.search(r'- \d{2}-[A-Z]{3}-\d{4}(.+)$',
176                             timetable_data[2]).group(1)
177        else:
178            name = timetable_data[2]
179
180        section_type = self._section_type_dct[
181            'O' if re.match(r'ONLINE COURSE', timetable_data[3]) else
182            re.match(r'[LBICR]', timetable_data[3]).group(0)]
183
184        modality = (self._modality_dct[timetable_data[4]] if
185                    timetable_data[4] in self._modality_dct else None)
186
187        class_dct = defaultdict(set)
188        for day in [self._day_dct[d] for d in timetable_data[8].split()]:
189            if day == Day.ARRANGED:
190                continue
191            class_dct[day].add((timetable_data[9], timetable_data[10],
192                                timetable_data[11]))
193        if (extra_class_data is not None and
194                extra_class_data[4] == '* Additional Times *'):
195            for day in [self._day_dct[d] for d in extra_class_data[8].split()]:
196                class_dct[day].add((extra_class_data[9], extra_class_data[10],
197                                    extra_class_data[11]))
198        class_dct = dict(class_dct)
199
200        self._course_data = {'year': year, 'semester': semester,
201                             'crn': timetable_data[0][:5],
202                             'subject': subject,
203                             'code': code,
204                             'name': name,
205                             'section_type': section_type,
206                             'modality': modality,
207                             'credit_hours': timetable_data[5],
208                             'capacity': timetable_data[6],
209                             'professor': timetable_data[7],
210                             'schedule': class_dct}
211
212    def __str__(self):
213        return ''.join(
214            f'{d}: {self._course_data[d]}, ' for d in self._course_data)[:-2]

270def get_crn(year: str, semester: Semester, crn: str) -> Course:
271    """Fetches a course with the provided CRN from the timetable.
272
273    Args:
274            year:
275                    string representing the year in which the course is taking place.
276            semester:
277                    Semester object representing the semester in which the course is
278                    taking place.
279            crn:
280                    string representing the Course Request Number of the course.
281
282    Returns:
283            A `Course` if there is a course with the provided CRN in the
284            provided year and semester, otherwise None.
285    """
286
287    crn_search = search_timetable(year, semester, crn=crn)
288    return crn_search[0] if crn_search else None

291def get_semesters() -> Set[Tuple[str, str]]:
292    """Fetches the semesters listed in the timetable.
293
294    Returns:
295            A set of length-2 tuples representing the semesters listed in the
296            timetable. The first element of each tuple is the semester, and the
297            second element of each tuple is the year.
298    """
299
300    semester_dct = {'Spring': Semester.SPRING, 'Summer': Semester.SUMMER,
301                    'Fall': Semester.FALL, 'Winter': Semester.WINTER}
302    return set((semester_dct[m.group(1)], m.group(2)) for m in re.finditer(
303        r'<OPTION VALUE="\d{6}">([A-Z][a-z]+) (\d+)<\/OPTION>',
304        _make_request(request_type='GET')))

307def get_subjects() -> Set[Tuple[str, str]]:
308    """Fetches the course subjects listed in the timetable.
309
310    Returns:
311            A set of length-2 tuples representing the course subjects listed in the
312            timetable. The first element of each tuple is the abbreviation of the
313            subject name, and the second element of each tuple is is the full
314            subject name.
315    """
316
317    return set((m.group(1), m.group(2)) for m in
318               re.finditer(r'\("([A-Z]+) - (.+?)"',
319                           _make_request(request_type='GET')))

322def search_timetable(year: str, semester: Semester,
323                     campus: Campus = Campus.BLACKSBURG,
324                     pathway: Pathway = Pathway.ALL, subject: str = '',
325                     section_type: SectionType = SectionType.ALL,
326                     code: str = '', crn: str = '',
327                     status: Status = Status.ALL,
328                     modality: Modality = Modality.ALL) -> List[Course]:
329    """Performs a search of the timetable with the given arguments.
330
331    Args:
332            year:
333                    string representing the year in which the course is taking place.
334            semester:
335                    `Semester` representing the semester in which the course is taking
336                    place.
337            campus:
338                    `Campus` representing the campus at which the course is taking
339                    place.
340            pathway:
341                    `Pathway` representing the Pathway/CLE designation of the course.
342            subject:
343                    string representing the subject of the course.
344            section_type:
345                    `SectionType` representing the section type of the course.
346            code:
347                    string representing the code number of the course.
348            crn:
349                    string representing the Course Request Number of the course.
350            status:
351                    `Status` representing whether the course needs to have open spots
352                    or not.
353            modality:
354                    `Modality` representing the modality of the course.
355
356    Returns:
357            List of courses returned by the search.
358    """
359
360    term_year = ((str(int(year) - 1) if semester == Semester.WINTER else year)
361                 + semester.value)
362    subject = '%' if subject == '' else subject
363    request = _make_request(request_type='POST',
364                            request_data={'CAMPUS': campus,
365                                          'TERMYEAR': term_year,
366                                          'CORE_CODE': pathway,
367                                          'subj_code': subject,
368                                          'SCHDTYPE': section_type,
369                                          'CRSE_NUMBER': code,
370                                          'crn': crn,
371                                          'open_only': status,
372                                          'sess_code': modality})
373    if request == '':
374        return []
375
376    request_data = read_html(request)[4]
377    course_list = []
378    for i in range(1, request_data.shape[0]):
379        if isinstance(request_data.iloc[i][0], str):
380            course_list.append(Course(year, semester, request_data.iloc[i],
381                                      request_data.iloc[i + 1] if request_data.shape[0] > i + 1 else None))
382    return course_list

385def _make_request(request_type: str,
386                  request_data: Dict[str, str] = None) -> str:
387    url = 'https://apps.es.vt.edu/ssb/HZSKVTSC.P_ProcRequest'
388    if request_type == 'POST':
389        for r in request_data:
390            request_data[r] = (request_data[r].value if
391                               issubclass(type(request_data[r]), Enum) else
392                               request_data[r])
393        request = requests.post(url, request_data)
394
395        if 'THERE IS AN ERROR WITH YOUR REQUEST' in request.text:
396            raise InvalidRequestException('The search parameters provided were invalid')
397
398        if 'There was a problem with your request' in request.text:
399            if 'NO SECTIONS FOUND FOR THIS INQUIRY' in request.text:
400                return ''
401            course_not_found_message = re.search(r'<b class=red_msg><li>(.+)</b>', request.text).group(1)
402            raise InvalidSearchException(course_not_found_message)
403        return request.text
404    elif request_type == 'GET':
405        return requests.get(url).text
406    
407    raise ValueError('Invalid request type')
